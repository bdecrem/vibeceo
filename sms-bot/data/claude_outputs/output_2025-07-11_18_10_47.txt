    <script>
        let partyData = [];

        async function loadData() {
            try {
                const response = await fetch(`/api/form/submissions?token=PARTY_ADMIN_7PM&app_id=PARTY_SIGNUP_TABLE`);
                const result = await response.json();
                if (result.error) throw new Error(result.error);
                
                partyData = result.submissions;
                updateDashboard();
            } catch (error) {
                console.error('Error:', error);
                alert('Failed to load guest list');
            }
        }

        function updateDashboard() {
            const tbody = document.getElementById('guestList');
            tbody.innerHTML = '';
            
            document.getElementById('totalGuests').textContent = partyData.length;
            
            const lastHourGuests = partyData.filter(guest => {
                const submitTime = new Date(guest.submission_time);
                return (Date.now() - submitTime) <= 3600000;
            }).length;
            document.getElementById('lastHour').textContent = lastHourGuests;

            partyData.forEach(guest => {
                const row = document.createElement('tr');
                row.innerHTML = `
                    <td>${guest.submission_data.name}</td>
                    <td>${guest.submission_data.email}</td>
                    <td>${guest.submission_data.phone}</td>
                    <td>${new Date(guest.submission_time).toLocaleString()}</td>
                `;
                tbody.appendChild(row);
            });
        }

        function exportToCSV() {
            const csvContent = 'Name,Email,Phone,Signup Time\n' + partyData.map(guest => 
                `${guest.submission_data.name},${guest.submission_data.email},${guest.submission_data.phone},${new Date(guest.submission_time).toLocaleString()}`
            ).join('\n');

            const blob = new Blob([csvContent], { type: 'text/csv' });
            const url = URL.createObjectURL(blob);
            const a = document.createElement('a');
            a.href = url;
            a.download = 'party_guest_list.csv';
            a.click();
        }

        // Load initial data
        loadData();

        // Refresh every minute
        setInterval(loadData, 60000);
    </script>
