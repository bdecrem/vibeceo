<script>
  var dpr = window.devicePixelRatio || 1;
  var controlsHeight = 120;
  var canvas = document.getElementById('game-canvas');
  var ctx = canvas.getContext('2d');
  var cw, ch, width, height;

  function resizeCanvas() {
    width = window.innerWidth;
    height = window.innerHeight - controlsHeight;
    canvas.width = width * dpr;
    canvas.height = height * dpr;
    canvas.style.width = width + 'px';
    canvas.style.height = height + 'px';
    ctx.scale(dpr, dpr);
    cw = width;
    ch = height;
  }

  window.addEventListener('resize', resizeCanvas);
  resizeCanvas();

  var gamePhase = 'start';
  var playerScore = 0;
  var aiScore = 0;
  var winScore = 5;

  var paddleWidth = 15;
  var paddleHeight = 100;
  var ballSize = 15;
  var playerY = ch / 2 - paddleHeight / 2;
  var aiY = ch / 2 - paddleHeight / 2;
  var ballX = cw / 2;
  var ballY = ch / 2;
  var ballSpeedX = 5;
  var ballSpeedY = 3;
  var playerSpeed = 0;
  var aiSpeed = 4;
  var lastTime = 0;
  var frameRate = 1000 / 60;

  document.getElementById('start-button').addEventListener('click', function() {
    document.getElementById('start-screen').style.display = 'none';
    document.getElementById('game-screen').style.display = 'flex';
    gamePhase = 'game';
    resetBall();
    playerScore = 0;
    aiScore = 0;
    updateScore();
  });

  document.getElementById('restart-button').addEventListener('click', function() {
    document.getElementById('end-screen').style.display = 'none';
    document.getElementById('game-screen').style.display = 'flex';
    gamePhase = 'game';
    resetBall();
    playerScore = 0;
    aiScore = 0;
    updateScore();
  });

  function resetBall() {
    ballX = cw / 2;
    ballY = ch / 2;
    ballSpeedX = -ballSpeedX;
    ballSpeedY = (Math.random() * 4) - 2;
  }

  function updateScore() {
    document.getElementById('player-score').textContent = playerScore;
    document.getElementById('ai-score').textContent = aiScore;
  }

  function update() {
    if (gamePhase !== 'game') return;

    playerY += playerSpeed;
    if (playerY < 0) playerY = 0;
    if (playerY > ch - paddleHeight) playerY = ch - paddleHeight;

    var aiCenter = aiY + paddleHeight / 2;
    if (aiCenter < ballY - 15) {
      aiY += aiSpeed;
    } else if (aiCenter > ballY + 15) {
      aiY -= aiSpeed;
    }
    if (aiY < 0) aiY = 0;
    if (aiY > ch - paddleHeight) aiY = ch - paddleHeight;

    ballX += ballSpeedX;
    ballY += ballSpeedY;

    if (ballY < 0 || ballY > ch - ballSize) {
      ballSpeedY = -ballSpeedY;
    }

    if (ballX < paddleWidth) {
      if (ballY + ballSize > playerY && ballY < playerY + paddleHeight) {
        var deltaY = ballY - (playerY + paddleHeight / 2);
        ballSpeedX = -ballSpeedX * 1.05;
        ballSpeedY = deltaY * 0.2;
      } else {
        aiScore++;
        updateScore();
        if (aiScore >= winScore) {
          endGame(false);
        } else {
          resetBall();
        }
      }
    }

    if (ballX > cw - paddleWidth - ballSize) {
      if (ballY + ballSize > aiY && ballY < aiY + paddleHeight) {
        var deltaY = ballY - (aiY + paddleHeight / 2);
        ballSpeedX = -ballSpeedX * 1.05;
        ballSpeedY = deltaY * 0.2;
      } else {
        playerScore++;
        updateScore();
        if (playerScore >= winScore) {
          endGame(true);
        } else {
          resetBall();
        }
      }
    }
  }

  function endGame(playerWon) {
    gamePhase = 'end';
    document.getElementById('game-screen').style.display = 'none';
    document.getElementById('end-screen').style.display = 'flex';
    document.getElementById('result-text').textContent = playerWon ? "You Win! üéâ" : "AI Wins! ü§ñ";
    document.getElementById('final-player').textContent = playerScore;
    document.getElementById('final-ai').textContent = aiScore;
  }

  function draw() {
    ctx.clearRect(0, 0, cw, ch);
    ctx.fillStyle = '#fff';
    
    ctx.fillRect(0, playerY, paddleWidth, paddleHeight);
    ctx.fillRect(cw - paddleWidth, aiY, paddleWidth, paddleHeight);
    
    ctx.beginPath();
    ctx.arc(ballX + ballSize/2, ballY + ballSize/2, ballSize/2, 0, Math.PI * 2);
    ctx.fill();
    
    for (var i = 0; i < ch; i += 20) {
      ctx.fillRect(cw/2 - 2, i, 4, 10);
    }
  }

  function gameLoop(timestamp) {
    var elapsed = timestamp - lastTime;
    if (elapsed > frameRate) {
      update();
      draw();
      lastTime = timestamp;
    }
    requestAnimationFrame(gameLoop);
  }

  document.getElementById('up-button').addEventListener('touchstart', function(e) {
    e.preventDefault();
    playerSpeed = -7;
  });
  document.getElementById('down-button').addEventListener('touchstart', function(e) {
    e.preventDefault();
    playerSpeed = 7;
  });
  document.getElementById('up-button').addEventListener('touchend', function() {
    playerSpeed = 0;
  });
  document.getElementById('down-button').addEventListener('touchend', function() {
    playerSpeed = 0;
  });

  document.getElementById('up-button').addEventListener('mousedown', function() {
    playerSpeed = -7;
  });
  document.getElementById('down-button').addEventListener('mousedown', function() {
    playerSpeed = 7;
  });
  document.addEventListener('mouseup', function() {
    playerSpeed = 0;
  });

  document.addEventListener('keydown', function(e) {
    if (e.key === 'ArrowUp') playerSpeed = -7;
    if (e.key === 'ArrowDown') playerSpeed = 7;
  });
  document.addEventListener('keyup', function() {
    playerSpeed = 0;
  });

  requestAnimationFrame(gameLoop);
</script>
