            <input type="text"

                   placeholder="Describe your dream song..."
                   aria-label="Music prompt input">
            
            <div class="checkbox-wrap">
                <input type="checkbox" 
                       id="instrumental" 
                       checked>
                <label for="instrumental">Instrumental Only</label>
            </div>
            
            <button id="generate">Generate Music ðŸŽµ</button>

    <script>
        const generateBtn = document.querySelector('#generate');
        const promptInput = document.querySelector('.prompt-input');
        const instrumentalCheck = document.querySelector('#instrumental');
        const status = document.querySelector('.status');
        const audioPlayer = document.querySelector('#audioPlayer');

        async function generateMusic(prompt, isInstrumental = true) {
            try {
                const finalPrompt = isInstrumental && !prompt.toLowerCase().includes('vocal') 
                    ? `${prompt}, instrumental, no vocals` 
                    : prompt;
                
                const response = await fetch('/api/sonauto', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ prompt: finalPrompt })
                });
                
                const data = await response.json();
                return await pollForCompletion(data.task_id);
                
            } catch (error) {
                console.error('Generation failed:', error);
                return null;
            }
        }

        async function pollForCompletion(taskId) {
            const maxAttempts = 60;
            let attempts = 0;
            
            while (attempts < maxAttempts) {
                try {
                    const response = await fetch(`/api/sonauto?taskId=${taskId}`);
                    const result = await response.json();
                    
                    const status = typeof result === 'string' ? result : result.status;
                    
                    if (status === 'SUCCESS' || status === 'COMPLETED') {
                        const audioUrl = result.song_paths?.[0] || result.url;
                        return { success: true, audioUrl, data: result };
                    }
                    
                    if (status === 'FAILED') {
                        return { success: false, error: 'Generation failed' };
                    }
                    
                    await new Promise(resolve => setTimeout(resolve, 2000));
                    attempts++;
                    
                } catch (error) {
                    console.error('Polling error:', error);
                    attempts++;
                }
            }
            
            return { success: false, error: 'Generation timed out' };
        }

        generateBtn.addEventListener('click', async () => {
            const prompt = promptInput.value.trim();
            
            if (!prompt) {
                promptInput.classList.add('shake');
                setTimeout(() => promptInput.classList.remove('shake'), 500);
                return;
            }

            generateBtn.disabled = true;
            status.textContent = 'ðŸŽµ Creating your music... (about 30 seconds)';
            
            const result = await generateMusic(prompt, instrumentalCheck.checked);
            
            if (result?.success) {
                status.textContent = 'âœ¨ Music ready! Hit play below';
                audioPlayer.src = result.audioUrl;
            } else {
                status.textContent = 'âŒ Oops! Something went wrong. Try again?';
            }
            
            generateBtn.disabled = false;
        });

        promptInput.addEventListener('keypress', (e) => {
            if (e.key === 'Enter') {
                generateBtn.click();
            }
        });
    </script>
