    <script>
        let submissionsData = [];

        async function loadSubmissions() {
            try {
                const response = await fetch(`/api/form/submissions?token=PARTY_ADMIN_7PM&app_id=PARTY_SIGNUP_TABLE`);
                const result = await response.json();
                
                if (result.error) throw new Error(result.error);
                
                submissionsData = result.submissions;
                updateStats();
                renderTable();
                
            } catch (error) {
                console.error('Error:', error);
                alert('Failed to load submissions');
            }
        }

        function updateStats() {
            document.getElementById('totalGuests').textContent = submissionsData.length;
            
            const lastHour = submissionsData.filter(sub => {
                return new Date(sub.timestamp) > new Date(Date.now() - 3600000);
            }).length;
            document.getElementById('lastHour').textContent = lastHour;
            
            document.getElementById('conversionRate').textContent = '85%';
        }

        function renderTable() {
            const tbody = document.getElementById('submissionsTable');
            tbody.innerHTML = '';
            
            submissionsData.forEach(sub => {
                const tr = document.createElement('tr');
                tr.innerHTML = `
                    <td>${sub.submission_data.name}</td>
                    <td>${sub.submission_data.email}</td>
                    <td>${sub.submission_data.phone}</td>
                    <td>${new Date(sub.timestamp).toLocaleString()}</td>
                `;
                tbody.appendChild(tr);
            });
        }

        function exportToCSV() {
            const csvContent = 'Name,Email,Phone,Timestamp\n' + 
                submissionsData.map(row => 
                    `${row.submission_data.name},${row.submission_data.email},${row.submission_data.phone},${new Date(row.timestamp).toLocaleString()}`
                ).join('\n');
            
            const blob = new Blob([csvContent], { type: 'text/csv' });
            const url = URL.createObjectURL(blob);
            const a = document.createElement('a');
            a.href = url;
            a.download = 'party_guest_list.csv';
            a.click();
        }

        // Initial load
        loadSubmissions();

        // Refresh every 30 seconds
        setInterval(loadSubmissions, 30000);
    </script>
