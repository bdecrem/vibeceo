    <script>
        const ADMIN_TOKEN = 'PARTY_ADMIN_TOKEN_2024';
        let rsvpData = [];

        async function loadRSVPs() {
            try {
                const response = await fetch(`/api/form/submissions?token=${ADMIN_TOKEN}`);
                const result = await response.json();
                if (result.error) throw new Error(result.error);
                
                rsvpData = result.submissions;
                updateStats();
                updateTable();
            } catch (error) {
                console.error('Error:', error);
                alert('Failed to load RSVPs');
            }
        }

        function updateStats() {
            const totalRsvps = rsvpData.length;
            const totalGuests = rsvpData.reduce((acc, curr) => acc + parseInt(curr.submission_data.guests), 0);
            const soloRiders = rsvpData.filter(r => r.submission_data.guests === '0').length;

            document.getElementById('totalRsvps').textContent = totalRsvps;
            document.getElementById('totalGuests').textContent = totalGuests + totalRsvps;
            document.getElementById('soloRiders').textContent = soloRiders;
        }

        function updateTable() {
            const tbody = document.getElementById('rsvpList');
            tbody.innerHTML = '';

            rsvpData.forEach(rsvp => {
                const row = document.createElement('tr');
                row.innerHTML = `
                    <td>${rsvp.submission_data.name}</td>
                    <td>${rsvp.submission_data.email}</td>
                    <td>+${rsvp.submission_data.guests}</td>
                    <td>${new Date(rsvp.timestamp).toLocaleString()}</td>
                `;
                tbody.appendChild(row);
            });
        }

        function exportCSV() {
            const csvContent = 'Name,Email,Guests,Timestamp\n' + rsvpData.map(row => 
                `${row.submission_data.name},${row.submission_data.email},${row.submission_data.guests},${new Date(row.timestamp).toLocaleString()}`
            ).join('\n');

            const blob = new Blob([csvContent], { type: 'text/csv' });
            const url = URL.createObjectURL(blob);
            const a = document.createElement('a');
            a.href = url;
            a.download = 'party_rsvps.csv';
            a.click();
        }

        function refreshData() {
            loadRSVPs();
        }

        // Initial load
        loadRSVPs();

        // Auto refresh every 5 minutes
        setInterval(loadRSVPs, 300000);
    </script>
