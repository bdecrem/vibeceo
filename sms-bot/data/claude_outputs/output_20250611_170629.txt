    <script>
        const item = document.getElementById('item');
        const compostBin = document.getElementById('compost-bin');
        const trashBin = document.getElementById('trash-bin');
        const tips = document.getElementById('tips');

        let draggableItem = null;

        item.addEventListener('dragstart', function (e) {
            draggableItem = this;
            setTimeout(() => this.style.display = 'none', 0);
        });

        item.addEventListener('dragend', function (e) {
            setTimeout(() => {
                this.style.display = 'block';
                draggableItem = null;
            }, 0);
        });

        [compostBin, trashBin].forEach(bin => {
            bin.addEventListener('dragover', function (e) {
                e.preventDefault();
            });

            bin.addEventListener('dragenter', function (e) {
                e.preventDefault();
            });

            bin.addEventListener('dragleave', function (e) {});

            bin.addEventListener('drop', function (e) {
                if (draggableItem) {
                    if (this.id === 'compost-bin') {
                        tips.textContent = "Great! üçé can go to compost! Reduce those food scraps!";
                    } else {
                        tips.textContent = "Oops! üçé doesn't belong in trash. Try composting!";
                    }
                    tips.classList.remove('hidden');
                    setTimeout(() => {
                        tips.classList.add('hidden');
                    }, 2000);
                }
            });
        });
    </script>
