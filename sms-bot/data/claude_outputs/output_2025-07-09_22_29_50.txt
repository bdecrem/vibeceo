    <script>
        let submissions = []

        async function loadSubmissions() {
            try {
                const urlParams = new URLSearchParams(window.location.search)
                const token = urlParams.get('token')
                const appSlug = window.location.pathname.split('/').pop().replace('admin-', '')
                
                if (!token) {
                    alert('Missing authentication token')
                    return
                }
                
                const response = await fetch(`/api/admin-submissions?token=${token}&app=${appSlug}`)
                const result = await response.json()
                
                if (!response.ok) throw new Error(result.error || 'Failed to load submissions')
                
                submissions = result.submissions || []
                updateDashboard()
            } catch (error) {
                console.error('Error:', error)
                alert('Failed to load guest list')
            }
        }

        function updateDashboard() {
            const tbody = document.getElementById('guestList')
            tbody.innerHTML = ''
            
            submissions.forEach(sub => {
                const tr = document.createElement('tr')
                tr.innerHTML = `
                    <td>${sub.submission_data.name}</td>
                    <td>${sub.submission_data.email}</td>
                    <td>${new Date(sub.created_at).toLocaleString()}</td>
                `
                tbody.appendChild(tr)
            })

            document.getElementById('totalRSVPs').textContent = submissions.length
            
            const lastHour = submissions.filter(sub => {
                return (new Date() - new Date(sub.created_at)) < 3600000
            }).length
            document.getElementById('lastHourRSVPs').textContent = lastHour
        }

        function exportCSV() {
            const csvContent = 'Name,Email,RSVP Time\n' + submissions.map(row => 
                `${row.submission_data.name},${row.submission_data.email},${new Date(row.created_at).toLocaleString()}`
            ).join('\n')
            
            const blob = new Blob([csvContent], { type: 'text/csv' })
            const url = URL.createObjectURL(blob)
            const a = document.createElement('a')
            a.href = url
            a.download = 'party_guest_list.csv'
            a.click()
        }

        window.addEventListener('load', loadSubmissions)
    </script>
