    <script>
        const supabase = window.supabase.createClient('YOUR_SUPABASE_URL', 'YOUR_SUPABASE_ANON_KEY');

        async function loadRSVPs() {
            try {
                const { data, error } = await supabase.from('wtaf_submissions')
                    .select('*')
                    .eq('app_id', 'senior_center_party')
                    .order('created_at', { ascending: false });

                if (error) throw error;

                const tbody = document.getElementById('rsvp-list');
                tbody.innerHTML = '';

                data.forEach(rsvp => {
                    const tr = document.createElement('tr');
                    tr.innerHTML = `
                        <td>${rsvp.submission_data.name}</td>
                        <td>${rsvp.submission_data.email}</td>
                        <td>${new Date(rsvp.created_at).toLocaleString()}</td>
                    `;
                    tbody.appendChild(tr);
                });

                document.getElementById('total-rsvps').textContent = data.length;

                window.rsvpData = data; // Store for CSV export
            } catch (error) {
                console.error('Error:', error);
                alert('Failed to load RSVPs');
            }
        }

        function exportCSV() {
            if (!window.rsvpData) return;

            const csvContent = 'Name,Email,Timestamp\n' + window.rsvpData.map(row => 
                `${row.submission_data.name},${row.submission_data.email},${new Date(row.created_at).toLocaleString()}`
            ).join('\n');

            const blob = new Blob([csvContent], { type: 'text/csv' });
            const url = URL.createObjectURL(blob);
            const a = document.createElement('a');
            a.href = url;
            a.download = 'senior_center_party_rsvps.csv';
            a.click();
        }

        // Initial load
        loadRSVPs();

        // Refresh every 30 seconds
        setInterval(loadRSVPs, 30000);
    </script>
