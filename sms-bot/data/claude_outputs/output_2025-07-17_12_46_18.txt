    <script>
        let currentUser = null;
        let pollingInterval;

        // Authentication System
        function initAuth() {
            // Authentication initialization code here
        }

        async function generateNewUser() {
            try {
                const slots = await checkAvailableSlots();
                
                if (slots.isFull) {
                    alert('ROBOT LAB AT CAPACITY! COME BACK LATER! 🚫');
                    return false;
                }
                
                const user = await generateUser();
                if (user && user.success) {
                    currentUser = {
                        userLabel: user.userLabel,
                        passcode: user.passcode,
                        participantId: user.participantId
                    };
                    
                    document.getElementById('new-user-info').innerHTML = 
                        `<strong>YOUR ENGINEER ID:</strong> ${currentUser.userLabel}<br><strong>LAB ACCESS CODE:</strong> ${currentUser.passcode}<br><em>SCREENSHOT THIS OR GET LOCKED OUT! 📸</em>`;
                    
                    return true;
                } else {
                    alert('ROBOT LAB MALFUNCTION! TRY AGAIN! 🔧');
                    return false;
                }
            } catch (error) {
                console.error('User generation error:', error);
                alert('SYSTEM CRASH! REBOOT AND TRY AGAIN! 💥');
                return false;
            }
        }

        async function registerNewUser() {
            if (!currentUser) {
                alert('GENERATE YOUR ENGINEER ID FIRST! 🎭');
                return;
            }
            
            try {
                const result = await registerUser(currentUser.userLabel, currentUser.passcode, currentUser.participantId);
                if (result && result.success) {
                    updateZadAuth(currentUser.userLabel, currentUser.participantId);
                    enterMainScreen();
                } else {
                    alert('REGISTRATION MALFUNCTION! TRY AGAIN! 💥');
                }
            } catch (error) {
                console.error('Registration error:', error);
                alert('REGISTRATION EXPLODED! TRY AGAIN! 💥');
            }
        }

        async function showNewUserScreen() {
            showScreen('new-user-screen');
            const success = await generateNewUser();
            if (!success) {
                showScreen('welcome-screen');
            }
        }

        async function loginReturningUser() {
            const selectedLabel = document.getElementById('user-label-select').value;
            const enteredPasscode = document.getElementById('returning-passcode').value.trim();
            
            if (!selectedLabel || selectedLabel === 'Select User') {
                alert('SELECT YOUR ENGINEER ID! 👻');
                return;
            }
            
            if (!enteredPasscode || enteredPasscode.length !== 4) {
                alert('4 DIGITS REQUIRED FOR LAB ACCESS! 🔢');
                return;
            }
            
            try {
                const result = await authenticateUser(selectedLabel, enteredPasscode);
                if (result && result.success) {
                    currentUser = {
                        userLabel: selectedLabel,
                        passcode: enteredPasscode,
                        participantId: result.user.participantId
                    };
                    updateZadAuth(currentUser.userLabel, currentUser.participantId);
                    enterMainScreen();
                } else {
                    alert('ACCESS DENIED! CHECK YOUR CREDENTIALS! ❌');
                }
            } catch (error) {
                console.error('Login error:', error);
                alert('LAB SYSTEMS DOWN! TRY AGAIN! 🌀');
            }
        }

        function showScreen(screenId) {
            document.querySelectorAll('.screen').forEach(screen => 
                screen.classList.remove('active'));
            document.getElementById(screenId).classList.add('active');
            
            if (screenId !== 'main-screen') {
                stopPolling();
            }
        }

        async function showReturningUserScreen() {
            showScreen('returning-user-screen');
            
            try {
                const slots = await checkAvailableSlots();
                const userSelect = document.getElementById('user-label-select');
                userSelect.innerHTML = '<option>Select User</option>';
                
                slots.usedLabels.forEach(label => {
                    const option = document.createElement('option');
                    option.value = label;
                    option.textContent = label;
                    userSelect.appendChild(option);
                });
                
                if (slots.usedLabels.length === 0) {
                    userSelect.innerHTML = '<option>No engineers registered</option>';
                }
            } catch (error) {
                console.error('Error loading users:', error);
                document.getElementById('user-label-select').innerHTML = '<option>Error loading engineers</option>';
            }
        }

        function startPolling() {
            pollingInterval = setInterval(loadLatestData, 2000);
        }

        function stopPolling() {
            if (pollingInterval) {
                clearInterval(pollingInterval);
                pollingInterval = null;
            }
        }

        async function loadLatestData() {
            try {
                const features = await load('robot_feature');
                const robotName = await load('robot_name');
                
                // Update robot name
                if (robotName && robotName.length > 0) {
                    document.getElementById('robot-name').textContent = robotName[0].name;
                } else {
                    document.getElementById('robot-name').textContent = 'UNNAMED ROBOT';
                }

                // Update features list
                const featuresList = document.getElementById('features-list');
                let featuresHTML = '<h3>INSTALLED FEATURES:</h3>';
                
                features.forEach(feature => {
                    featuresHTML += `
                        <div class="robot-feature">
                            <span class="user-tag">${feature.userLabel}</span>
                            ${feature.feature}
                        </div>
                    `;
                });
                
                featuresList.innerHTML = featuresHTML;

                // Disable input if user already added a feature
                const userFeature = features.find(f => f.userLabel === currentUser.userLabel);
                if (userFeature) {
                    document.getElementById('feature-input').disabled = true;
                    document.getElementById('feature-input').placeholder = 'You\'ve already added your feature!';
                }
            } catch (error) {
                console.error('Error loading data:', error);
            }
        }

        async function addFeature() {
            const featureInput = document.getElementById('feature-input');
            const feature = featureInput.value.trim();
            
            if (!feature) {
                alert('ADD A FEATURE FIRST! 🔧');
                return;
            }

            try {
                await save('robot_feature', {
                    feature: feature,
                    userLabel: currentUser.userLabel,
                    timestamp: Date.now()
                });

                // If this is the first feature, generate a robot name
                const features = await load('robot_feature');
                if (features.length === 1) {
                    const robotNames = [
                        'CHAOS-TRON 3000',
                        'MEGA-GLITCH IX',
                        'QUANTUM DESTROYER',
                        'CYBER PHOENIX',
                        'NEON WRAITH'
                    ];
                    const randomName = robotNames[Math.floor(Math.random() * robotNames.length)];
                    await save('robot_name', { name: randomName });
                }

                featureInput.value = '';
                loadLatestData();
            } catch (error) {
                console.error('Error saving feature:', error);
                alert('FEATURE INSTALLATION FAILED! TRY AGAIN! 💥');
            }
        }

        function enterMainScreen() {
            showScreen('main-screen');
            document.getElementById('current-user-label').textContent = currentUser.userLabel;
            startPolling();
            loadLatestData();
        }

        function leaveApp() {
            stopPolling();
            showScreen('welcome-screen');
        }

        // Mouse parallax effect
        document.addEventListener('mousemove', (e) => {
            document.querySelectorAll('.floating-emoji').forEach((elem) => {
                const speed = elem.getAttribute('data-value') || 2;
                const x = (e.clientX * speed) / 100;
                const y = (e.clientY * speed) / 100;
                elem.style.transform = `translateX(${x}px) translateY(${y}px)`;
            });
        });

        // Easter egg
        console.log('🤖 Welcome to the Robot Lab! Type robot() for a surprise...');
        window.robot = function() {
            document.body.style.animation = 'rainbow 1s infinite';
            setTimeout(() => {
                document.body.style.animation = '';
                alert('🤖 QUANTUM CORE OVERLOAD! YOU\'VE UNLEASHED THE ROBOT REVOLUTION!');
            }, 3000);
        };

        // Initialize app
        initAuth();
    </script>
