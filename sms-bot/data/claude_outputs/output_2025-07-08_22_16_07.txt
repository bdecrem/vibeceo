        async function toggleGoal(index) {
            try {
                const { data: userData } = await supabase
                    .from('wtaf_zero_admin_collaborative')
                    .select('*')
                    .eq('app_id', APP_ID)
                    .eq('participant_id', currentUser.participantId)
                    .eq('action_type', 'join')
                    .single();

                if (!userData) return;

                let completedToday = userData.content_data.completedToday || [];
                let streak = userData.content_data.streak || 0;
                const lastCheck = userData.content_data.lastCheck;
                const today = new Date().toDateString();

                // Reset completedToday if it's a new day
                if (lastCheck !== today) {
                    completedToday = [];
                }

                // Toggle goal completion
                if (completedToday.includes(index)) {
                    completedToday = completedToday.filter(i => i !== index);
                } else {
                    completedToday.push(index);
                    createCelebration();
                }

                // Update streak
                if (completedToday.length === 3) {
                    if (lastCheck !== today) streak++;
                } else if (lastCheck !== today) {
                    streak = 0;
                }

                // Update database
                await supabase
                    .from('wtaf_zero_admin_collaborative')
                    .update({
                        content_data: {
                            completedToday,
                            streak,
                            lastCheck: today,
                            timestamp: Date.now()
                        }
                    })
                    .eq('id', userData.id);

                loadLatestData();
            } catch (error) {
                console.error('Error toggling goal:', error);
            }
        }
