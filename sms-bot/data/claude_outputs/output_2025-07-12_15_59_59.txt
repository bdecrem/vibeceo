        async function init() {
            username = prompt('Enter your cosmic identifier:') || 'Anonymous Stardust';
            loadEntries();

        async function saveEntry() {
            const input = document.getElementById('journal-input');
            if (!input.value.trim()) return;

            await save('journal_entry', {
                text: input.value,
                author: username,
                timestamp: Date.now()
            });

            input.value = '';
            loadEntries();
        }

        async function loadEntries() {
            const entries = await load('journal_entry');
            const container = document.getElementById('entries');
            
            container.innerHTML = entries.sort((a, b) => b.timestamp - a.timestamp)
                .map(entry => `
                    <div class="journal-entry">
                        <div>${entry.text}</div>
                        <div class="meta-info">
                            ‚úçÔ∏è ${entry.author} ‚Ä¢ ${new Date(entry.timestamp).toLocaleString()}

                `).join('');
        }

        function setupParallax() {
            document.addEventListener('mousemove', (e) => {
                document.querySelectorAll('.floating-emoji').forEach(emoji => {
                    const speed = emoji.getAttribute('data-value') || 2;
                    const x = (e.clientX * speed) / 100;
                    const y = (e.clientY * speed) / 100;
                    emoji.style.transform = `translate(${x}px, ${y}px)`;
                });
            });
        }
