        async function saveEntry() {
            const content = document.getElementById('new-entry').value;
            if (!content.trim()) return;

            const entry = {
                content,
                timestamp: new Date().toISOString()
            };

            await fetch('/api/zad/save', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({
                    app_id: APP_ID,
                    participant_id: 'user1',
                    action_type: 'journal_entry',
                    content_data: entry
                })
            });

            document.getElementById('new-entry').value = '';
            loadEntries();
        }

        async function loadEntries() {
            const response = await fetch('/api/zad/load?app_id=' + APP_ID + '&action_type=journal_entry');
            const data = await response.json();
            
            const container = document.getElementById('entries-container');
            container.innerHTML = '';

            data.forEach(item => {
                const entry = item.content_data;
                const div = document.createElement('div');
                div.className = 'journal-entry';
                div.innerHTML = `
                    <div class="entry-date">${new Date(entry.timestamp).toLocaleString()}</div>
                    <p>${entry.content}</p>
                `;
                container.appendChild(div);
            });
        }
