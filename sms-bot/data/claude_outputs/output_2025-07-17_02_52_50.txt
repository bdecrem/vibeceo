    <script>
        // Parallax effect
        document.addEventListener('mousemove', (e) => {
            const emojis = document.querySelectorAll('.floating-emoji');
            emojis.forEach(emoji => {
                const speed = emoji.getAttribute('data-value');
                const x = (e.clientX * speed) / 250;
                const y = (e.clientY * speed) / 250;
                emoji.style.transform = `translate(${x}px, ${y}px)`;
            });
        });

        // Color schemes
        const gradients = [
            'linear-gradient(45deg, #FF2E93, #3F88FF, #FF7C5C, #7C3AED)',
            'linear-gradient(45deg, #00FFC6, #57E2E5, #8CFFDA, #3F88FF)',
            'linear-gradient(45deg, #FF7C5C, #FFB178, #FFEAC2, #FF2E93)',
            'linear-gradient(45deg, #7C3AED, #A855F7, #D8B4FE, #FF61C7)'
        ];

        // Next button handler
        document.getElementById('nextBtn').addEventListener('click', () => {
            const randomGradient = gradients[Math.floor(Math.random() * gradients.length)];
            document.querySelector('.gradient-bg').style.background = randomGradient;
        });

        // Save button handler
        document.getElementById('saveBtn').addEventListener('click', () => {
            localStorage.setItem('savedDesign', document.querySelector('.gradient-bg').style.background);
            document.querySelector('.btn-container').style.display = 'none';
        });

        // Easter egg
        let wtafCode = '';
        document.addEventListener('keydown', (e) => {
            wtafCode += e.key.toLowerCase();
            if(wtafCode.includes('wtaf')) {
                document.body.style.animation = 'spin 1s linear';
                wtafCode = '';
            }
        });

        // Check for saved design
        window.onload = () => {
            const savedDesign = localStorage.getItem('savedDesign');
            if(savedDesign) {
                document.querySelector('.gradient-bg').style.background = savedDesign;
                document.querySelector('.btn-container').style.display = 'none';
            }
        };

        @keyframes spin {
            0% { transform: rotate(0deg); }
            100% { transform: rotate(360deg); }
        }
