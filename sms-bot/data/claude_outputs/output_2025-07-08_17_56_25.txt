        function renderGoals() {
            const container = document.getElementById('goals-container');
            container.innerHTML = '';
            
            userGoals.forEach((goal, index) => {
                const goalDiv = document.createElement('div');
                goalDiv.className = 'goal-checkbox';
                goalDiv.innerHTML = `
                    <div class="checkbox" onclick="toggleGoal(${index})"></div>
                    <span>${goal}</span>
                `;
                container.appendChild(goalDiv);
            });

        async function toggleGoal(index) {
            const { data: currentData } = await supabase
                .from('wtaf_zero_admin_collaborative')
                .select('*')
                .eq('app_id', APP_ID)
                .eq('participant_id', currentUser.participantId)
                .eq('action_type', 'join')
                .single();

            const completedToday = currentData.content_data.completedToday || [];
            const isCompleted = completedToday.includes(index);

            if (!isCompleted) {
                completedToday.push(index);
                celebrateGoal();
            } else {
                const idx = completedToday.indexOf(index);
                completedToday.splice(idx, 1);
            }

            await supabase
                .from('wtaf_zero_admin_collaborative')
                .update({
                    content_data: {
                        ...currentData.content_data,
                        completedToday,
                        lastCheckIn: Date.now()
                    }
                })
                .eq('id', currentData.id);

            await loadLatestData();
        }

            document.body.appendChild(celebration);
            setTimeout(() => celebration.remove(), 1000);
        }

        function updateGoalsDisplay(data) {
            const completedToday = data.completedToday || [];
            document.querySelectorAll('.goal-checkbox').forEach((checkbox, index) => {
                checkbox.classList.toggle('completed', completedToday.includes(index));
                checkbox.querySelector('.checkbox').textContent = 
                    completedToday.includes(index) ? '✓' : '';
            });

            // Update streak display
            const streak = data.streak || 0;
            document.getElementById('streak-display').textContent = 
                `Current Streak: ${streak} days 🔥`;

            // Update weekly progress
            const progress = (completedToday.length / 3) * 100;
            document.getElementById('weekly-progress').style.width = `${progress}%`;

            // Check for Sunday celebration

            if (today.getDay() === 0 && completedToday.length === 3) {

            }
