    <script>
        let currentUser = null;
        let pollingInterval;

        async function initPublicMode() {
            currentUser = await generateAnonymousUser();
            if (currentUser?.success) {
                updateZadAuth(currentUser.userLabel, currentUser.participantId);
                document.getElementById('current-user-label').textContent = currentUser.userLabel;
                startPolling();
                loadLatestData();
            }
        }

        function startPolling() {
            pollingInterval = setInterval(() => loadLatestData(), 2000);
        }

        async function loadLatestData() {
            const wishes = await load('birthday_wish');
            const sortedWishes = wishes.sort((a, b) => 
                new Date(b.created_at) - new Date(a.created_at)
            );
            
            const container = document.getElementById('wishes-container');
            container.innerHTML = sortedWishes.map(wish => `
                <div class="birthday-wish">
                    <div class="wish-message">${wish.message}</div>
                    <div class="wish-author">
                        ${wish.customName ? '‚ù§Ô∏è ' + wish.customName : 'üéà ' + wish.author}
                    </div>
                    <div class="wish-time">
                        ${new Date(wish.created_at).toLocaleString()}
                    </div>
                </div>
            `).join('');
        }

        async function submitWish(event) {
            event.preventDefault();
            
            const message = document.getElementById('message').value;
            const customName = document.getElementById('name').value.trim();
            
            await save('birthday_wish', {
                message,
                customName
            });

            // Clear form
            document.getElementById('message').value = '';
            document.getElementById('name').value = '';

            // Create emoji animation
            createEmojiRain();

            // Refresh data
            await loadLatestData();
        }

        function createEmojiRain() {
            const emojis = ['üéÇ', 'üéâ', 'üéà', 'üéÅ', 'üéä', 'üíù', 'üåü'];
            const container = document.createElement('div');
            container.className = 'emoji-rain';
            
            for (let i = 0; i < 20; i++) {
                const emoji = document.createElement('div');
                emoji.style.position = 'absolute';
                emoji.style.left = Math.random() * 100 + 'vw';
                emoji.style.top = -20 + 'px';
                emoji.style.fontSize = '20px';
                emoji.textContent = emojis[Math.floor(Math.random() * emojis.length)];
                
                // Animation
                emoji.animate([
                    { transform: 'translateY(0) rotate(0deg)' },
                    { transform: `translateY(${window.innerHeight}px) rotate(${Math.random() * 360}deg)` }
                ], {
                    duration: 1500 + Math.random() * 1000,
                    easing: 'linear'
                });

                container.appendChild(emoji);
            }

            document.body.appendChild(container);
            setTimeout(() => container.remove(), 2000);
        }

        // Initialize the app
        initPublicMode();
    </script>
