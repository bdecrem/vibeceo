    <script>
        let partyData = [];

        async function loadPartyData() {
            try {
                const response = await fetch(`/api/form/submissions?token=PARTY_ADMIN_TOKEN_7PM&app_id=PARTY_SIGNUP_TABLE`);
                const result = await response.json();
                if (result.error) throw new Error(result.error);
                
                partyData = result.submissions;
                updateDashboard();
            } catch (error) {
                console.error('Error:', error);
                alert('Failed to load party data');
            }
        }

        function updateDashboard() {
            const tbody = document.getElementById('guestList');
            tbody.innerHTML = '';

            partyData.forEach(guest => {
                const row = document.createElement('tr');
                row.innerHTML = `
                    <td>${guest.submission_data.name}</td>
                    <td>${guest.submission_data.email}</td>
                    <td>${guest.submission_data.phone}</td>
                    <td>${new Date(guest.created_at).toLocaleString()}</td>
                `;
                tbody.appendChild(row);
            });

            // Update stats
            document.getElementById('totalSignups').textContent = partyData.length;
            
            const lastHourSignups = partyData.filter(guest => {
                return new Date(guest.created_at) > new Date(Date.now() - 3600000);
            }).length;
            document.getElementById('lastHour').textContent = lastHourSignups;

            const conversion = ((partyData.length / 100) * 100).toFixed(1);
            document.getElementById('conversion').textContent = `${conversion}%`;
        }

        function exportCSV() {
            const csvContent = 'Name,Email,Phone,Signup Time\n' + partyData.map(guest => 
                `${guest.submission_data.name},${guest.submission_data.email},${guest.submission_data.phone},${new Date(guest.created_at).toLocaleString()}`
            ).join('\n');

            const blob = new Blob([csvContent], { type: 'text/csv' });
            const url = URL.createObjectURL(blob);
            const a = document.createElement('a');
            a.href = url;
            a.download = 'party_guest_list.csv';
            a.click();
        }

        // Load initial data
        loadPartyData();

        // Refresh data every 30 seconds
        setInterval(loadPartyData, 30000);
    </script>
