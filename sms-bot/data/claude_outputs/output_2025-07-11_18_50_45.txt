    <script>
        let submissions = [];

        async function loadSubmissions() {
            try {
                const response = await fetch(`/api/form/submissions?token=PARTY_ADMIN_7PM&app_id=PARTY_SIGNUP_TABLE`);
                const result = await response.json();
                if (result.error) throw new Error(result.error);
                
                submissions = result.submissions;
                updateTable(submissions);
                updateStats();
            } catch (error) {
                console.error('Error:', error);
                alert('Failed to load submissions');
            }
        }

        function updateTable(data) {
            const table = document.getElementById('signupsTable');
            table.innerHTML = data.map(row => `
                <tr>
                    <td>${row.submission_data.name}</td>
                    <td>${row.submission_data.email}</td>
                    <td>${row.submission_data.phone}</td>
                    <td>${new Date(row.timestamp).toLocaleString()}</td>
                </tr>
            `).join('');
        }

        function updateStats() {
            document.getElementById('totalSignups').textContent = submissions.length;
            
            const lastHour = submissions.filter(s => 
                new Date(s.timestamp) > new Date(Date.now() - 3600000)
            ).length;
            document.getElementById('lastHour').textContent = lastHour;

            document.getElementById('conversionRate').textContent = 
                Math.round((submissions.length / 100) * 100) + '%';
        }

        function exportToCSV() {
            const csvContent = 'Name,Email,Phone,Signup Time\n' + 
                submissions.map(row => 
                    `${row.submission_data.name},${row.submission_data.email},${row.submission_data.phone},${new Date(row.timestamp).toLocaleString()}`
                ).join('\n');

            const blob = new Blob([csvContent], { type: 'text/csv' });
            const url = URL.createObjectURL(blob);
            const a = document.createElement('a');
            a.href = url;
            a.download = 'party_guest_list.csv';
            a.click();
        }

        document.getElementById('searchBar').addEventListener('input', (e) => {
            const searchTerm = e.target.value.toLowerCase();
            const filtered = submissions.filter(s => 
                s.submission_data.name.toLowerCase().includes(searchTerm) ||
                s.submission_data.email.toLowerCase().includes(searchTerm) ||
                s.submission_data.phone.includes(searchTerm)
            );
            updateTable(filtered);
        });

        // Initial load
        loadSubmissions();

        // Refresh data every 30 seconds
        setInterval(loadSubmissions, 30000);
    </script>
