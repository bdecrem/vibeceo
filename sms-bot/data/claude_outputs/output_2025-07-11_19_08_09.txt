    <script>
        let partyData = [];

        async function loadPartyData() {
            try {
                const response = await fetch(`/api/form/submissions?token=PARTY_ADMIN_7PM&app_id=PARTY_SIGNUP_TABLE`);
                const result = await response.json();
                if (result.error) throw new Error(result.error);
                
                partyData = result.submissions;
                updateDashboard();
                
            } catch (error) {
                console.error('Error:', error);
                alert('Failed to load party data');
            }
        }

        function updateDashboard() {
            document.getElementById('totalRSVPs').textContent = partyData.length;
            document.getElementById('pendingConfirmations').textContent = partyData.filter(guest => !guest.confirmed).length;
            document.getElementById('specialRequests').textContent = partyData.filter(guest => guest.submission_data.message).length;

            const guestList = document.getElementById('guestList');
            guestList.innerHTML = partyData.map(guest => `
                <tr>
                    <td>${guest.submission_data.name}</td>
                    <td>${guest.submission_data.email}</td>
                    <td>${guest.submission_data.phone || 'N/A'}</td>
                    <td>${guest.submission_data.message || 'None'}</td>
                    <td>${new Date(guest.timestamp).toLocaleString()}</td>
                </tr>
            `).join('');
        }

        function exportCSV() {
            const csvContent = 'Name,Email,Phone,Special Requests,Timestamp\n' + 
                partyData.map(guest => `${guest.submission_data.name},${guest.submission_data.email},${guest.submission_data.phone || ''},${guest.submission_data.message || ''},${new Date(guest.timestamp).toLocaleString()}`).join('\n');
            
            const blob = new Blob([csvContent], { type: 'text/csv' });
            const url = URL.createObjectURL(blob);
            const a = document.createElement('a');
            a.href = url;
            a.download = 'party_guest_list.csv';
            a.click();
        }

        // Load data on page load
        loadPartyData();

        // Refresh data every 30 seconds
        setInterval(loadPartyData, 30000);
    </script>
