        async function addTodo() {

            const text = input.value.trim();
            
            if (!text) {
                alert('EMPTY TASKS ARE VOID TASKS üëª');
                return;
            }

            try {
                const { data: existingTodos } = await supabase
                    .from('wtaf_zero_admin_collaborative')
                    .select('*')
                    .eq('app_id', APP_ID)
                    .eq('action_type', 'todos')
                    .single();

                const todos = existingTodos?.content_data?.todos || [];
                todos.push({
                    text,
                    completed: false,
                    createdBy: currentUser.userLabel,
                    timestamp: Date.now()
                });

                if (existingTodos) {
                    await supabase
                        .from('wtaf_zero_admin_collaborative')
                        .update({
                            content_data: { todos }
                        })
                        .eq('app_id', APP_ID)
                        .eq('action_type', 'todos');
                } else {
                    await supabase
                        .from('wtaf_zero_admin_collaborative')
                        .insert({
                            app_id: APP_ID,
                            participant_id: currentUser.participantId,
                            action_type: 'todos',
                            participant_data: {
                                userLabel: currentUser.userLabel
                            },
                            content_data: { todos }
                        });
                }

                input.value = '';
                loadLatestData();
            } catch (error) {
                console.error('Error adding todo:', error);
                alert('TASK CREATION FAILED IN THE VORTEX üå™Ô∏è');

                    .from('wtaf_zero_admin_collaborative')
