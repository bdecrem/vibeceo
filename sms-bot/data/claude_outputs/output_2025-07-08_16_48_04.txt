        async function submitIdea() {
            const ideaText = document.getElementById('idea-input').value.trim();

                alert('YOUR IDEA IS TOO ETHEREAL! MAKE IT REAL! 💭');

                        timestamp: Date.now(),

            } catch (error) {

            }

        async function addReaction(ideaId) {
            try {
                const { data: idea } = await supabase
                    .from('wtaf_zero_admin_collaborative')
                    .select('*')
                    .eq('id', ideaId)
                    .single();

                const newReactions = (idea.content_data.reactions || 0) + 1;
                
                await supabase
                    .from('wtaf_zero_admin_collaborative')
                    .update({
                        content_data: {
                            ...idea.content_data,
                            reactions: newReactions
                        }
                    })
                    .eq('id', ideaId);

                await loadLatestData();
            } catch (error) {
                console.error('Reaction error:', error);
                alert('REACTION OVERLOAD! SYSTEM COOLING! 🌡️');
            }
        }

        async function addComment(ideaId) {
            const commentText = prompt('DROP YOUR THOUGHTS 💭');
            if (!commentText) return;

            try {
                const { data: idea } = await supabase
                    .from('wtaf_zero_admin_collaborative')
                    .select('*')
                    .eq('id', ideaId)
                    .single();

                const comments = idea.content_data.comments || [];
                comments.push({
                    user: currentUser.userLabel,
                    text: commentText,
                    timestamp: Date.now()
                });

                await supabase
                    .from('wtaf_zero_admin_collaborative')
                    .update({
                        content_data: {
                            ...idea.content_data,
                            comments: comments
                        }
                    })
                    .eq('id', ideaId);

                await loadLatestData();
            } catch (error) {
                console.error('Comment error:', error);
                alert('COMMENT VORTEX UNSTABLE! TRY AGAIN! 🌪️');
            }
        }

        function sortIdeas(method) {
            currentSort = method;
            document.querySelectorAll('.sort-btn').forEach(btn => btn.classList.remove('active'));
            event.target.classList.add('active');
            loadLatestData();
        }

        async function loadLatestData() {
            try {
                const { data: ideas } = await supabase
                    .from('wtaf_zero_admin_collaborative')
                    .select('*')
                    .eq('app_id', APP_ID)
                    .eq('action_type', 'idea');

                let sortedIdeas = [...ideas];
                if (currentSort === 'reactions') {
                    sortedIdeas.sort((a, b) => (b.content_data.reactions || 0) - (a.content_data.reactions || 0));
                } else {
                    sortedIdeas.sort((a, b) => a.content_data.category.localeCompare(b.content_data.category));
                }

                const container = document.getElementById('ideas-container');
                container.innerHTML = '';

                sortedIdeas.forEach(idea => {
                    const card = document.createElement('div');
                    card.className = 'idea-card';
                    card.innerHTML = `
                        <div class="category-tag">${idea.content_data.category}</div>
                        <p>${idea.content_data.idea}</p>
                        <small>By ${idea.participant_data.userLabel}</small>
                        <div class="reactions">
                            <button class="fire-btn" onclick="addReaction(${idea.id})">🔥</button>
                            <span class="fire-count">${idea.content_data.reactions || 0}</span>
                            <button onclick="addComment(${idea.id})">💭 Comment</button>
                        </div>
                        <div class="comments-section">
                            ${(idea.content_data.comments || []).map(comment => `
                                <div class="comment">
                                    <strong>${comment.user}:</strong> ${comment.text}
                                </div>
                            `).join('')}
                        </div>
                    `;
                    container.appendChild(card);
                });
            } catch (error) {
                console.error('Load error:', error);
                alert('DATA STREAM INTERRUPTED! RECALIBRATING! 📡');
            }
        }

            pollingInterval = setInterval(async () => {

            if (pollingInterval) {
                clearInterval(pollingInterval);
                pollingInterval = null;
