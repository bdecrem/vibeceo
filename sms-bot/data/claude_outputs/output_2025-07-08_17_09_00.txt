        function showScreen(screenId) {
            document.querySelectorAll('.screen').forEach(screen => 
                screen.classList.remove('active'));
            document.getElementById(screenId).classList.add('active');
            
            if (screenId !== 'main-screen') {
                stopPolling();
            }
        }

        function enterMainScreen() {
            showScreen('main-screen');
            document.getElementById('current-user-label').textContent = currentUser.userLabel;
            initializeHabitGrid();
            startPolling();
            loadLatestData();
        }

        // Habit Tracking Functions
        function initializeHabitGrid() {
            const grid = document.getElementById('habit-grid');
            grid.innerHTML = '<div class="habit-header">Habits</div>';
            
            // Add day headers
            ['Mon', 'Tue', 'Wed', 'Thu', 'Fri', 'Sat', 'Sun'].forEach(day => {
                grid.innerHTML += `<div class="habit-header">${day}</div>`;
            });

            // Add habit rows
            habits.forEach(habit => {
                grid.innerHTML += `<div class="habit-header">${habit}</div>`;
                for (let i = 0; i < 7; i++) {
                    grid.innerHTML += `<div class="habit-cell" onclick="toggleHabit('${habit}', ${i})" id="${habit}-${i}">‚ùå</div>`;
                }
            });
        }

        async function toggleHabit(habit, day) {
            if (!habitData[habit]) habitData[habit] = new Array(7).fill(false);
            habitData[habit][day] = !habitData[habit][day];
            
            try {
                await supabase.from('wtaf_zero_admin_collaborative').upsert({
                    app_id: APP_ID,
                    participant_id: currentUser.participantId,
                    action_type: 'update',
                    participant_data: {
                        userLabel: currentUser.userLabel
                    },
                    content_data: {
                        habits: habitData,
                        timestamp: Date.now()
                    }
                });

                updateHabitDisplay();
                checkAchievements();
            } catch (error) {
                console.error('Error updating habit:', error);
                alert('HABIT UPDATE FAILED, UNIVERSE GLITCHING üåÄ');
            }
        }

        function updateHabitDisplay() {
            habits.forEach(habit => {
                if (habitData[habit]) {
                    habitData[habit].forEach((completed, day) => {
                        const cell = document.getElementById(`${habit}-${day}`);
                        cell.textContent = completed ? '‚úÖ' : '‚ùå';
                        cell.className = `habit-cell ${completed ? 'completed' : ''}`;
                    });
                }
            });

            updateProgressChart();
        }

        function checkAchievements() {
            habits.forEach(habit => {
                if (habitData[habit]) {
                    let streak = 0;
                    let maxStreak = 0;
                    
                    habitData[habit].forEach(completed => {
                        if (completed) {
                            streak++;
                            maxStreak = Math.max(maxStreak, streak);
                        } else {
                            streak = 0;
                        }
                    });

                    if (maxStreak >= 5) {
                        const achievementId = {
                            'Water': 'hydration-achievement',
                            'Sleep': 'sleep-achievement',
                            'Screen Time': 'screen-achievement'
                        }[habit];
                        
                        document.getElementById(achievementId).classList.add('unlocked');
                    }
                }
            });
        }

        function updateProgressChart() {
            const chart = document.getElementById('progress-chart');
            chart.innerHTML = '';
            
            habits.forEach((habit, index) => {
                if (habitData[habit]) {
                    const completed = habitData[habit].filter(done => done).length;
                    const percentage = (completed / 7) * 100;
                    
                    const bar = document.createElement('div');
                    bar.className = 'chart-bar';
                    bar.style.height = `${percentage}%`;
                    bar.style.left = `${(index * 33) + 15}%`;
                    bar.title = `${habit}: ${percentage.toFixed(1)}%`;
                    
                    chart.appendChild(bar);
                }
            });
        }

        // Polling for Updates

        function stopPolling() {
            if (pollingInterval) {

                pollingInterval = null;

            try {

                    .eq('app_id', APP_ID)
                    .eq('participant_id', currentUser.participantId)

                    habitData = data[0].content_data.habits || {};
                    updateHabitDisplay();
                    checkAchievements();
                }
            } catch (error) {

            }
