    <script>
        let guestData = [];

        async function loadGuestList() {
            try {
                const response = await fetch(`/api/form/submissions?token=PARTY_ADMIN_7PM&app_id=PARTY_SIGNUP_TABLE`);
                const result = await response.json();
                if (result.error) throw new Error(result.error);
                
                guestData = result.submissions;
                updateStats();
                renderGuestList();
            } catch (error) {
                console.error('Error:', error);
                alert('Failed to load guest list');
            }
        }

        function updateStats() {
            const total = guestData.length;
            const confirmed = guestData.filter(g => g.status === 'confirmed').length;
            
            document.getElementById('totalGuests').textContent = total;
            document.getElementById('confirmedGuests').textContent = confirmed;
            document.getElementById('pendingGuests').textContent = total - confirmed;
        }

        function renderGuestList() {
            const tbody = document.getElementById('guestList');
            tbody.innerHTML = guestData.map(guest => `
                <tr>
                    <td>${guest.submission_data.name}</td>
                    <td>${guest.submission_data.email}</td>
                    <td>${guest.submission_data.phone}</td>
                    <td>${new Date(guest.created_at).toLocaleString()}</td>
                    <td>
                        <div class="status-badge ${guest.status === 'confirmed' ? 'status-confirmed' : 'status-pending'}">
                            ${guest.status}
                        </div>
                    </td>
                </tr>
            `).join('');
        }

        function exportToCSV() {
            const csvContent = 'Name,Email,Phone,Signup Time,Status\n' + 
                guestData.map(guest => `${guest.submission_data.name},${guest.submission_data.email},${guest.submission_data.phone},${new Date(guest.created_at).toLocaleString()},${guest.status}`).join('\n');
            
            const blob = new Blob([csvContent], { type: 'text/csv' });
            const url = URL.createObjectURL(blob);
            const a = document.createElement('a');
            a.href = url;
            a.download = 'party_guest_list.csv';
            a.click();
        }

        // Initial load
        loadGuestList();

        // Refresh every 30 seconds
        setInterval(loadGuestList, 30000);
    </script>
