        async function submitIdea() {
            const ideaInput = document.getElementById('new-idea-input');
            const ideaText = ideaInput.value.trim();
            
            if (!ideaText) {
                alert('EMPTY IDEAS = EMPTY DREAMS üò¥');

            try {
