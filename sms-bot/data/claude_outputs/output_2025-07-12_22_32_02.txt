    <script>
        let username;
        const emojis = ['âœ¨', 'ðŸ’­', 'ðŸŒ™', 'âš¡', 'ðŸŒˆ'];

        async function init() {
            username = prompt('Enter your name to start journaling:') || 'Anonymous Soul';
            createFloatingEmojis();
            loadEntries();
            setInterval(loadEntries, 5000);
        }

        function createFloatingEmojis() {
            emojis.forEach((emoji, index) => {
                const element = document.createElement('div');
                element.className = 'floating-emoji';
                element.textContent = emoji;
                element.style.left = `${Math.random() * 90}vw`;
                element.style.top = `${Math.random() * 90}vh`;
                element.style.animationDelay = `${index * 0.5}s`;
                document.body.appendChild(element);
            });
        }

        async function saveEntry() {
            const text = document.getElementById('entry-text').value;
            if (!text.trim()) return;

            await save('journal_entry', {
                text,
                author: username,
                timestamp: Date.now()
            });

            document.getElementById('entry-text').value = '';
            loadEntries();
        }

        async function loadEntries() {
            const entries = await load('journal_entry');
            const entriesHTML = entries
                .sort((a, b) => b.timestamp - a.timestamp)
                .map(entry => `
                    <div class="entry">
                        <div class="entry-header">
                            <span>${entry.author}</span>
                            <span>${new Date(entry.timestamp).toLocaleString()}</span>
                        </div>
                        ${entry.text}
                    </div>
                `).join('');
            
            document.getElementById('entries').innerHTML = entriesHTML;
        }

        document.getElementById('entry-text').addEventListener('keydown', (e) => {
            if (e.key === 'Enter' && e.ctrlKey) {
                saveEntry();
            }
        });

        // Easter egg
        let wtafTyped = '';
        document.addEventListener('keydown', (e) => {
            wtafTyped += e.key.toLowerCase();
            if (wtafTyped.includes('wtaf')) {
                document.body.style.animation = 'gradient 5s ease infinite';
                alert('ðŸŽ‰ You found the secret! Extra chaos mode activated!');
                wtafTyped = '';
            }
        });

        init();
    </script>
