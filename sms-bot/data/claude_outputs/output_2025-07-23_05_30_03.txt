```html
<!DOCTYPE html>
<html lang="en">
<head>
<meta charset="UTF-8">
<meta name="viewport" content="width=device-width, initial-scale=1.0">
<title>ZAD APP - Shared Street</title>
<style>
  body, html { margin: 0; padding: 0; overflow: hidden; font-family: Arial, sans-serif; }
  #gameCanvas { display: block; background: linear-gradient(to bottom, #87CEEB, #f0f0f0); }
  #inputForm { position: fixed; bottom: 20px; left: 50%; transform: translateX(-50%); background: rgba(255, 255, 255, 0.8); padding: 10px; border-radius: 8px; }
  #buildingInput { width: 200px; padding: 5px; }
  #addButton { padding: 5px 10px; }
</style>
</head>
<body>
<canvas id="gameCanvas"></canvas>
<div id="inputForm">
  <input type="text" id="buildingInput" placeholder="Type a building...">
  <button id="addButton">Add Building</button>
</div>
<script>
  var canvas = document.getElementById('gameCanvas');
  var ctx = canvas.getContext('2d', { alpha: false });
  var buildings = [];
  var scrollOffset = 0;
  var buildingWidth = 100;
  var buildingHeight = 200;
  var lastTime = 0;
  var gameSpeed = 16;
  
  canvas.width = window.innerWidth * devicePixelRatio;
  canvas.height = window.innerHeight * devicePixelRatio;
  canvas.style.width = window.innerWidth + 'px';
  canvas.style.height = window.innerHeight + 'px';
  ctx.scale(devicePixelRatio, devicePixelRatio);

  function generateImage(buildingType) {
    var colors = ['#FF6347', '#4682B4', '#32CD32', '#FFD700', '#FF69B4'];
    var color = colors[Math.floor(Math.random() * colors.length)];
    return { type: buildingType, color: color };
  }

  function addBuilding() {
    var input = document.getElementById('buildingInput').value.trim();
    if (input) {
      var newBuilding = generateImage(input);
      buildings.push(newBuilding);
      document.getElementById('buildingInput').value = '';
    }
  }

  document.getElementById('addButton').addEventListener('click', addBuilding);

  function update() {
    scrollOffset += 1;
    if (scrollOffset > buildingWidth) {
      scrollOffset = 0;
      buildings.shift();
    }
  }

  function draw() {
    ctx.clearRect(0, 0, window.innerWidth, window.innerHeight);
    for (var i = 0; i < buildings.length; i++) {
      var x = i * buildingWidth - scrollOffset;
      ctx.fillStyle = buildings[i].color;
      ctx.fillRect(x, window.innerHeight - buildingHeight, buildingWidth, buildingHeight);
      ctx.fillStyle = '#000';
      ctx.fillText(buildings[i].type, x + 10, window.innerHeight - buildingHeight + 20);
    }
  }

  function gameLoop(currentTime) {
    if (currentTime - lastTime > gameSpeed) {
      lastTime = currentTime;
      update();
    }
    draw();
    requestAnimationFrame(gameLoop);
  }

  requestAnimationFrame(gameLoop);
</script>
</body>
</html>
```
