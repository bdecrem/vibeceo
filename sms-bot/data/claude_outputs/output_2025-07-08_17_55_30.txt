                document.getElementById('goals-tracker').style.display = 'block';
                updateGoalsDisplay();

        function updateGoalsDisplay() {
            const today = new Date().toLocaleDateString();
            document.querySelector('.date-display').textContent = today;

            const goalsList = document.getElementById('goals-list');
            goalsList.innerHTML = '';

            userGoals.forEach((goal, index) => {
                const container = document.createElement('div');
                container.className = 'goal-container';

                const label = document.createElement('span');
                label.textContent = goal;

                const checkbox = document.createElement('div');
                checkbox.className = 'checkbox';
                checkbox.setAttribute('data-index', index);
                checkbox.onclick = () => toggleGoal(index);

                if (todayProgress?.[index]) {
                    checkbox.classList.add('checked');
                    checkbox.textContent = '✓';
                }

                container.appendChild(label);
                container.appendChild(checkbox);
                goalsList.appendChild(container);
            });

            updateProgress();
        }

        async function toggleGoal(index) {
            const checkbox = document.querySelector(`.checkbox[data-index="${index}"]`);
            const isChecked = checkbox.classList.contains('checked');
            
            if (!isChecked) {
                checkbox.classList.add('checked');
                checkbox.textContent = '✓';
                
                // Celebration animation
                const emoji = CELEBRATION_EMOJIS[Math.floor(Math.random() * CELEBRATION_EMOJIS.length)];
                const celebration = document.createElement('div');
                celebration.textContent = emoji;
                celebration.style.position = 'absolute';
                celebration.style.left = `${checkbox.offsetLeft}px`;
                celebration.style.top = `${checkbox.offsetTop}px`;
                celebration.style.animation = 'celebrate 0.5s ease-out';
                document.body.appendChild(celebration);
                setTimeout(() => celebration.remove(), 500);
            } else {
                checkbox.classList.remove('checked');
                checkbox.textContent = '';
            }

                    .from('wtaf_zero_admin_collaborative')

                    .from('wtaf_zero_admin_collaborative')
                    .update({
                        content_data: {
                            ...userRecord.content_data,
                            progress: progress
                        }
                    })
                    .eq('id', userRecord.id);

                updateProgress();

                    } else {
                        showWeeklyMessage(weekProgress);
                    }
                }
            } catch (error) {
                console.error('Error updating progress:', error);
                alert('PROGRESS UPDATE FAILED 💔 TRY AGAIN!');
            }
        }
