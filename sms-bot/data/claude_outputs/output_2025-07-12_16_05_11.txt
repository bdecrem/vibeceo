        let username;

        async function init() {
            username = prompt('Enter your cosmic identity:') || 'Anonymous Star Child';
            positionEmojis();

        function positionEmojis() {
            document.querySelectorAll('.floating-emoji').forEach((emoji, index) => {
                emoji.style.left = `${Math.random() * 90}vw`;
                emoji.style.top = `${Math.random() * 90}vh`;
                emoji.style.animation = `float ${5 + index}s ease-in-out infinite`;
            });
        }

        document.addEventListener('mousemove', (e) => {
            document.querySelectorAll('.floating-emoji').forEach(emoji => {
                const speed = emoji.getAttribute('data-value') || 2;
                const x = (e.clientX * speed) / 250;
                const y = (e.clientY * speed) / 250;
                emoji.style.transform = `translate(${x}px, ${y}px)`;
            });
        });

        async function saveEntry() {
            const text = document.getElementById('entry-input').value.trim();
            if (!text) return;

            await save('journal_entry', {
                text,
                author: username,
                timestamp: Date.now()
            });

            document.getElementById('entry-input').value = '';
            loadEntries();
        }

        async function loadEntries() {
            const entries = await load('journal_entry');
            const entriesHTML = entries
                .sort((a, b) => b.timestamp - a.timestamp)
                .map(entry => `
                    <div class="entry">
                        <div class="entry-header">
                            <span>${entry.author}</span>
                            <span>${new Date(entry.timestamp).toLocaleString()}</span>
                        </div>
                        <div class="entry-content">${entry.text}</div>
                    </div>
                `).join('');
            
            document.getElementById('entries').innerHTML = entriesHTML;
        }
