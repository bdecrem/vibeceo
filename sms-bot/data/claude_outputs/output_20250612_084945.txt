```html
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Crossword Puzzle</title>
    <style>
        body {
            display: flex;
            justify-content: center;
            align-items: center;
            height: 100vh;
            background-color: #f0f8ff;
            font-family: 'Comic Neue', sans-serif;
        }
        .crossword {
            display: grid;
            grid-template-columns: repeat(9, 40px);
            grid-template-rows: repeat(9, 40px);
            gap: 3px;
        }
        .cell {
            width: 40px;
            height: 40px;
            background-color: #fffacd;
            border-radius: 5px;
            text-align: center;
            font-size: 1.2em;
            display: flex;
            justify-content: center;
            align-items: center;
        }
        .black-cell {
            background-color: #000;
        }
        .cell input {
            width: 90%;
            height: 90%;
            border: none;
            background: transparent;
            text-align: center;
            font-size: 1em;
        }
        .cell input:focus {
            outline: none;
        }
    </style>
</head>
<body>
    <div class="crossword" id="crossword">
    </div>
    <script>
        const crossword = [
            ['B', 'L', 'A', 'C', 'K', '', '', '', ''],
            ['L', '', '', '', 'P', '', '', '', ''],
            ['A', '', '', 'C', 'A', 'R', 'D', '', ''],
            ['C', '', '', '', 'O', '', '', '', ''],
            ['K', 'N', 'I', 'G', 'H', 'T', '', '', ''],
            ['', '', '', '', '', '', '', '', ''],
            ['', '', '', 'F', 'L', 'U', 'S', 'H', ''],
            ['', '', '', '', 'A', '', '', '', ''],
            ['', '', '', '', '', '', '', '', '']
        ];

        const crosswordElement = document.getElementById('crossword');

        function createCell(content, black = false) {
            const cell = document.createElement('div');
            cell.classList.add('cell');
            if (black) {
                cell.classList.add('black-cell');
            } else {
                const input = document.createElement('input');
                input.maxLength = 1;
                input.value = content;
                input.oninput = function () {
                    this.value = this.value.toUpperCase();
                };
                cell.appendChild(input);
            }
            return cell;
        }
        
        crossword.forEach(row => {
            row.forEach(cellContent => {
                crosswordElement.appendChild(createCell(cellContent, cellContent === ''));
            });
        });
    </script>
</body>
</html>
```
