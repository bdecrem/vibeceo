    <script>
        // Load submissions
        async function loadSubmissions() {
            try {
                const response = await fetch(`/api/form/submissions?token=PARTY_ADMIN_7PM&app_id=PARTY_SIGNUP_TABLE`);
                const result = await response.json();
                if (result.error) throw new Error(result.error);
                
                const data = result.submissions;
                updateStats(data);
                updateTable(data);
            } catch (error) {
                console.error('Error:', error);
                alert('Failed to load submissions');
            }
        }

        function updateStats(data) {
            document.getElementById('totalSignups').textContent = data.length;
            
            const lastHour = data.filter(entry => {
                return (new Date() - new Date(entry.timestamp)) < 3600000;
            }).length;
            document.getElementById('lastHour').textContent = lastHour;
            
            document.getElementById('conversionRate').textContent = 
                Math.round((data.length / (data.length + 20)) * 100) + '%';
        }

        function updateTable(data) {
            const tbody = document.getElementById('guestList');
            tbody.innerHTML = data.map(entry => `
                <tr>
                    <td>${entry.submission_data.name}</td>
                    <td>${entry.submission_data.email}</td>
                    <td>${entry.submission_data.phone}</td>
                    <td>${new Date(entry.timestamp).toLocaleString()}</td>
                </tr>
            `).join('');
        }

        function exportToCSV() {
            const data = Array.from(document.getElementById('guestList').rows).map(row => {
                return Array.from(row.cells).map(cell => cell.textContent);
            });
            
            const csvContent = 'Name,Email,Phone,Signup Time\n' + 
                data.map(row => row.join(',')).join('\n');
            
            const blob = new Blob([csvContent], { type: 'text/csv' });
            const url = URL.createObjectURL(blob);
            const a = document.createElement('a');
            a.href = url;
            a.download = 'party_guest_list.csv';
            a.click();
        }

        // Initial load
        loadSubmissions();
        
        // Refresh every 30 seconds
        setInterval(loadSubmissions, 30000);
    </script>
