```html
<!DOCTYPE html>
<html lang="en">
<head>
<meta charset="UTF-8">
<meta name="viewport" content="width=device-width, initial-scale=1.0">
<title>Collaborative City Builder</title>
<style>
  body, html { margin: 0; padding: 0; overflow: hidden; font-family: Arial, sans-serif; }
  #canvas { display: block; background: linear-gradient(to bottom, #87CEEB, #FFFFFF); }
  #form { position: fixed; bottom: 20px; left: 50%; transform: translateX(-50%); }
  #buildingInput { padding: 10px; font-size: 16px; }
  #addButton { padding: 10px; font-size: 16px; cursor: pointer; }
</style>
</head>
<body>
<canvas id="canvas"></canvas>
<div id="form">
  <input type="text" id="buildingInput" placeholder="Type a building (e.g., Coffee Shop)" />
  <button id="addButton">Add Building</button>
</div>
<script>
  var canvas = document.getElementById('canvas');
  var ctx = canvas.getContext('2d', { alpha: false });
  var buildings = [];
  var scrollOffset = 0;
  var buildingWidth = 150;
  var buildingHeight = 300;
  var speed = 2;

  function resizeCanvas() {
    canvas.width = window.innerWidth * devicePixelRatio;
    canvas.height = window.innerHeight * devicePixelRatio;
    canvas.style.width = window.innerWidth + 'px';
    canvas.style.height = window.innerHeight + 'px';
    ctx.scale(devicePixelRatio, devicePixelRatio);
  }

  function generateImage(buildingType) {
    var img = document.createElement('canvas');
    img.width = buildingWidth;
    img.height = buildingHeight;
    var imgCtx = img.getContext('2d');
    imgCtx.fillStyle = '#333';
    imgCtx.fillRect(0, 0, buildingWidth, buildingHeight);
    imgCtx.fillStyle = '#FFF';
    imgCtx.font = '20px Arial';
    imgCtx.fillText(buildingType, 10, 50);
    return img;
  }

  function addBuilding(type) {
    var image = generateImage(type);
    buildings.push({ type: type, image: image });
  }

  function draw() {
    ctx.clearRect(0, 0, window.innerWidth, window.innerHeight);
    for (var i = 0; i < buildings.length; i++) {
      var x = i * buildingWidth - scrollOffset;
      ctx.drawImage(buildings[i].image, x, window.innerHeight - buildingHeight);
    }
  }

  function update() {
    scrollOffset += speed;
    if (scrollOffset > buildings.length * buildingWidth) {
      scrollOffset = 0;
    }
  }

  function gameLoop() {
    update();
    draw();
    requestAnimationFrame(gameLoop);
  }

  document.getElementById('addButton').addEventListener('click', function() {
    var buildingType = document.getElementById('buildingInput').value.trim();
    if (buildingType) {
      addBuilding(buildingType);
      document.getElementById('buildingInput').value = '';
    }
  });

  window.addEventListener('resize', resizeCanvas);
  resizeCanvas();
  gameLoop();
</script>
</body>
</html>
```
